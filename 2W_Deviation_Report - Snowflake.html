@st.cache_data(ttl=300)
def load_firm_data() -> pd.DataFrame:
session = get_active_session()
sql = """
WITH last_two AS (
SELECT DISTINCT "Load_Timestamp"
FROM HDSN_FIRM
ORDER BY "Load_Timestamp" DESC
FETCH FIRST 2 ROWS ONLY
)
SELECT *
FROM HDSN_FIRM
WHERE "Load_Timestamp" IN (SELECT "Load_Timestamp" FROM last_two)
ORDER BY "Load_Timestamp" DESC;
"""
df = session.sql(sql).to_pandas()

# Standardize columns (Updated mapping to catch "LINE REQ. DATE")
col_mapping = {
"LOAD_TIMESTAMP": "Load_Timestamp",
"LINE REQ. DATE": "Line_Req_Date",
"LINE_REQ_DATE": "Line_Req_Date",
"CLIENT_ID": "Client_Id",
"H_D_PART": "H-D Part",
"QTY": "Qty",
"ISSUER": "Issuer"
}
df.columns = [col_mapping.get(str(c).upper(), c) for c in df.columns]

if "Load_Timestamp" in df.columns:
df["Load_Timestamp"] = pd.to_datetime(df["Load_Timestamp"])
if "Line_Req_Date" in df.columns:
if df["Line_Req_Date"].dtype == object:

df["Line_Req_Date"] = pd.to_datetime(df["Line_Req_Date"], errors="coerce")

return df
